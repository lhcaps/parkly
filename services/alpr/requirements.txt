# FastAPI + YOLOv8 + PaddleOCR ALPR Service
# GPU-accelerated license plate recognition for Parkly parking system
#
# Python: 3.12+ supported. Works on Python 3.10/3.11 as well.

# ── Server ────────────────────────────────────────────────────────────────────
uvicorn[standard]==0.32.1
fastapi==0.115.6
python-multipart==0.0.20   # FastAPI file upload support

# ── Model inference ─────────────────────────────────────────────────────────────
# YOLOv8 (Ultralytics) — downloads yolov8n.pt on first run.
ultralytics==8.3.52

# PaddlePaddle: use GPU build if you have CUDA, otherwise CPU fallback.

#   CPU: pip install paddlepaddle==3.3.0
paddlepaddle==2.5.2

# PaddleOCR 2.10.0 — Windows CPU compatible (works with paddlepaddle 2.5.x)
paddleocr==2.10.0

# ── Utilities ────────────────────────────────────────────────────────────────
numpy==1.26.4
Pillow==10.4.0
opencv-python-headless==4.10.0.84   # headless = no GUI dep, works in server env

# Fallback OCR if PaddleOCR fails on Windows CPU (oneDNN/PIR errors)
# Requires Tesseract OCR to be installed on the system.

# Then set TESSERACT_CMD env var to the tesseract executable path if not in PATH

# ── HTTP client (for image URL fetching) ─────────────────────────────────────
